   'Select C.*,'
      'O.Descricao Centro_Custo,'
      'T.Nome Terceiro,'
      'P.Descricao Categoria_Profissional,'
      'P.Descricao_Categoria Categoria_Funcoes,'
      'U.Grupo,'
      'E.Nome Entidade,'
      'Y.Descricao Pais,'
      'R.Descricao Carreira,'
      ''
      '(Select Count(*) From Recrutamento_Linhas L'
      'Where L.Ano_Recrutamento = C.Ano_Recrutamento'

        'And L.Numero_Recrutamento = C.Numero_Recrutamento) Total_Candida' +
        'tos,'
      ''
      '(Select Count(*) From Recrutamento_Linhas L'
      'Where L.Ano_Recrutamento = C.Ano_Recrutamento'
      'And L.Numero_Recrutamento = C.Numero_Recrutamento'
      'And L.Data_Entrevista Is Not Null) Total_Entrevistados,'
      ''
      '(Select Count(*) From Recrutamento_Linhas L'
      'Where L.Ano_Recrutamento = C.Ano_Recrutamento'
      'And L.Numero_Recrutamento = C.Numero_Recrutamento'
      'And L.Data_Analise_Curricular Is Not Null) Total_Acurricular,'
      ''
      '(Select Count(*) From Recrutamento_Linhas L'
      'Where L.Ano_Recrutamento = C.Ano_Recrutamento'
      'And L.Numero_Recrutamento = C.Numero_Recrutamento'
      'And L.Seleccionado='#39'S'#39') Total_Selecionados,'
      ''
      'C.Quantidade - (Select Count(*) From Recrutamento_Linhas L'
      'Where L.Ano_Recrutamento = C.Ano_Recrutamento'
      'And L.Numero_Recrutamento = C.Numero_Recrutamento'
      'And L.Seleccionado='#39'S'#39') Total_Em_Falta'
      ''
      'From Recrutamento_Cabecalho C'
      ''
      'Left Outer Join Centros_Custo O'
      'On O.Codigo_Centro_Custo = C.Codigo_Centro_Custo'
      ''
      'Left Outer Join Paises Y'
      'On Y.Codigo_Pais = O.Codigo_Pais'
      ''
      'Left Outer Join Terceiros E'
      'On E.Codigo_Terceiro = O.Codigo_Terceiro'
      ''
      'Left Outer Join Categorias_Profissionais P'
      'On P.Codigo_Categoria = C.Codigo_Categoria'
      ''
      'Left Outer Join Funcoes_Internas I'
      'On I.Codigo_Funcao_Interna = P.Codigo_Funcao_Interna'
      ''
      'Left Outer Join Empregados_Carreiras R'
      'On R.Codigo_Carreira = I.Codigo_Carreira'
      ''
      'Left Outer Join Utilizadores U'
      'On U.Username = C.Username_Responsavel'
      ''
      'Left Outer Join Terceiros T'
      'On T.Codigo_Terceiro = C.Codigo_Terceiro'
      ''
      'Where 1=2'
      ''
      'Order By C.Numero_Recrutamento'
